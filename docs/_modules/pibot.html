"""

This module contains the PiBOT class that incorporates the control,
motion, sensors, and outputs modules to create a fully functioning
robot. The PiBOT class contains attributes to track the robot position
and heading, determine its current state, perform lidar scans of the
surroundings, and work with the lidar data. The examples in the
following section demonstrate how to use the PiBOT attributes. The
motion, sensors, and outputs modules also contain many examples on how
to use their features from within an instance of the PiBOT class.

"""

from math import pi, cos, radians, sqrt
from utime import sleep_ms
from control import Control
from motion import Motion
from sensors import Whiskers, IrSensors, LidarSensor, IrRemote
from outputs import Buzzer, LEDs

    """
    Creates a top-level robot object to operate the PiBOT.

    ...

    Attributes
    ----------
    whiskers : Whiskers object
        Checks for left or right whisker contact.
    ir : IrSensors object
        Checks for left or right IR sensor detection.
    lidar : LidarSensor object
        Gets the current distance reading from the lidar sensor.
    remote : IrRemote object
        Gets commands from the IR remote control.
    leds : LEDs object
        Controls the left and right neopixel LEDs.
    buzzer : Buzzer object
        Controls the buzzer sound output.
    move : Motion object
        Controls the robot motion.
    lidar_dist : list
        The lidar distance values stored when scanning.
    lidar_angle : list
        The headings (angle values) when scanning with lidar.
    """

    def __init__(self):
        """Creates robot control attributes."""

        # OBJECTS NEEDED TO CREATE ROBOT
        self.whiskers = Whiskers()
        self.ir = IrSensors()
        self.lidar = LidarSensor()
        self.remote = IrRemote()
        self.leds = LEDs()
        self.buzzer = Buzzer()
        # pass objects into the Control class to allow access to attributes
        self._control = Control(leds=self.leds, buzzer=self.buzzer,
                               remote=self.remote)
        # pass instance of Control class into Motion class to access attributes
        self.move = Motion(self._control)
        # ATTRIBUTES FOR LIDAR SCAN
        self.lidar_dist = []
        self.lidar_angle = []

    @property
    def position(self):
        """Gets or set the current position as a list of [x, y]
        coordinates.

        Notes
        -----
        The global coordinate system is based on the position of the
        centerpoint between the robot's wheels when motion is first
        started or after resetting. The x axis points forward along the
        path, and the y axis points to the left wheel.

        Examples
        --------

        Create an instance of PiBOT.

        >>> from pibot import PiBOT
        >>> robot = PiBOT()

        Get the position.

        >>> robot.position
        [0, 0]

        Set the position.

        >>> robot.position = [7.07, -7.07]

        """

        # wait to avoid getting new value while _tracking() method is active
        while self._control._tracking_lock:
            continue
        return self._control._position

    @position.setter
    def position(self, value):
        # check for valid argument
        if not isinstance(value, list):
            return print('Error: position must be a list in the form [x, y]')
        if len(value) != 2:
            return print('Error: position must be a list of [x, y] values')
        if not all(isinstance(i, (int, float)) for i in value):
            return print('Error: x and y must be numeric values')
        # wait to avoid setting new value while _tracking() method is active
        while self._control._tracking_lock:
            continue
        self._control._position = value

    @property
    def heading(self):
        """Gets or set the current heading angle.

        Notes
        -----
        The heading angle is measured in degrees from the x axis with
        positive values going counterclockwise.

        Examples
        --------

        Create an instance of PiBOT.

        >>> from pibot import PiBOT
        >>> robot = PiBOT()

        Get the heading.

        >>> robot.heading
        89.54

        Set the heading.

        >>> robot.heading = 180

        """

        # wait to avoid setting new value while _tracking() method is active
        while self._control._tracking_lock:
            continue
        return self._control._heading * (180/pi)

    @heading.setter
    def heading(self, value):
        # check for valid argument
        if not isinstance(value, (int, float)):
            return print('Error: heading must be a numeric value in degrees')
        # wait to avoid setting new value while _tracking() method is active
        while self._control._tracking_lock:
            continue
        self._control._heading = value * (pi/180)

    @property
    def current_time(self):
        """Gets the current time in seconds as a read-only value.

        This is the time since an instance of the PiBOT class was
        created. The clock is reset when the .reset() method is called.

        """

        return self._control._t_current

    @property
    def motion_state(self):
        """Gets the current motion state of the robot.

        The motion states are: 'stop', 'pause', 'linear', 'turn', 'arc',
        and 'steer'.

        """

        return self._control._motion_state

    @property
    def busy(self):
        """Determines if the robot is busy executing motion commands.

        The robot is considered busy if protected motion is being
        commanded or if commands are waiting in the queue.

        """

        if self._control._protect or self._control._motion_queue:
            return True
        else:
            return False


[docs]    def reset(self):
        """Stops motion, exits thread, and reinitializes attributes."""

        self.move.pause()
        sleep_ms(15)
        self._control._thread_running = False
        # wait for thread to exit before reinitializing and starting new thread 
        sleep_ms(25)
        self._control.__init__(leds=self.leds, buzzer=self.buzzer,
                               remote=self.remote)



[docs]    def scan(self, angle, increment=2.5, filename=None):
        """Sweeps the robot through an angle and stores lidar data.

        Parameters
        ----------
        angle : int, float, maximum +/- 720
            A value in degrees to rotate the robot for a scan sweep.
        increment : int, float, default=2.5
            The interval in degrees at which to capture lidar data.
        filename : str, optional
            Name of the file for saving the lidar data to a CSV.
            
        Returns
        -------
        2-tuple of lists
            The corresponding angle and distance values, which can
            also be accessed through the .lidar_angle and .lidar_dist
            attributes.

        Notes
        -----
        The default angle increment of the lidar scan is set to 2.5
        degrees, which represents a good compromise between the speed
        and resolution during the turn motion. The actual angle stored
        in .lidar_angle varies because the data is collected dynamically
        while the robot turns. The lidar detects a field of view of
        about 10 degrees, meaning the data spaced at 2.5 degrees
        represents a higher resolution than the lidar can really discern
        and results in overlapping ranges of detection. 
        
        The optional filename is used to save a CSV file to the RP2040's
        memory, which can be accessed with Thonny and downloaded to the
        computer for analysis.

        Examples
        --------

        Create an instance of PiBOT.

        >>> from pibot import PiBOT
        >>> robot = PiBOT()

        Command a 360 degree clockwise scan. The returned lists are
        stored in local variables as shown; however, it is not
        necessary to use the returned values because they can also
        be accessed directly with the .lidar_dist and .lidar_angle
        attributes.

        >>> angle, distance = robot.scan(-360)

        Command a 180 degree counterclockwise scan and store the data
        to a file called 'test_data'. The data is not saved to local
        variables. In the output, it is clear that the default angle
        increment of 2.5 degrees cannot be followed exactly but tracks
        fairly close to the desired step size.

        >>> robot.scan(180, filename='test_data')
        ([0.0, 2.713043, 6.052173, 8.869564, 10.01739, 12.52174,...

        """

        # check for valid argument
        if not isinstance(angle, (int, float)) or abs(angle) > 720:
            return print('Error: maximum magnitude of angle is +/-720 degrees')
        if not isinstance(increment, (int, float)) or increment > abs(angle):
            return print('Error: increment in degrees must be less than angle')
        if filename and not isinstance(filename, str):
            return print('Error: filename must be a string')
        # create an angle increment counter
        i = 0
        start_angle = self.heading
        # clear lidar data before a new scan
        self.lidar_dist.clear()
        self.lidar_angle.clear()
        # start a protected turn move at the specified angle
        self.move.turn(angle, protect=True)
        # wait for the turn to start before collecting data
        while self._control._motion_state != 'turn':
            continue
        # while the turn is active, collect data at the approximate increment
        while self._control._motion_state == 'turn':
            if ((angle > 0
                 and self.heading >= start_angle + increment*i)
                    or (angle < 0
                        and self.heading <= start_angle - increment*i)):
                self.lidar_dist.append(self.lidar.read())
                self.lidar_angle.append(self.heading)
                i += 1
        # create a CSV (comma-separated values) file and write a column header
        if filename:
            file = open(f'{filename}.csv', 'w')
            file.write('angle (degrees), distance (cm)\n')
            for i in range(len(self.lidar_angle)):
                file.write(f'{self.lidar_angle[i]}, {self.lidar_dist[i]}\n')
            file.close()
        return self.lidar_angle, self.lidar_dist



[docs]    def max_distance(self, angle, distance):
        """Finds the maximum value in the lidar scan data.

        Parameters
        ----------
        angle : list
            The lidar angle data
        distance : list
            The lidar distance data

        Returns
        -------
        2-tuple
            A pair comprising the heading angle and maximum distance.

        Notes
        -----

        Finds the maximum value and corresponsing angle in scan data. If
        there is more than one identical maximum value, the first one in
        the list will be returned. If there are values that are out of
        range (i.e., 132 cm), the method finds the block of 132 cm
        values in the distance list that is longest and returns the
        angle near the center of the block. If there is only one 132 cm
        value, that will be returned as the maximum. If there is more
        than one non-adjacent single 132 cm value, the first one in the
        list will be returned as the maximum.

        Examples
        --------

        Create an instance of PiBOT.

        >>> from pibot import PiBOT
        >>> robot = PiBOT()

        Command a 360 degree clockwise scan and use the returned values
        to find the maximum.

        >>> angle, distance = robot.scan(-360)
        >>> robot.max_distance(angle, distance)
        (-188.4522, 132)

        The arguments can also be the lidar data taken directly from the
        PiBOT attributes instead of local copies used in the previous
        example.

        >>> robot.scan(-360)
        ([0.0, -2.504347, -5.321739, -7.61739, -10.53913, -12.93913,...
        >>> robot.max_distance(robot.lidar_angle, robot.lidar_dist)
        (-80.34782, 91.0)

        The scan can also return its values directly to the max_distance
        method by using the unpacking operator (*) as shown.

        >>> robot.max_distance(*robot.scan(-360))
        (-78.78259, 90.0)

        """

        count = 0
        max_count = 0
        index = []
        # create index list of all out-of-range values in lidar data
        for i in range(distance.count(132)):
            if i == 0:
                index.append(distance.index(132))
            else:
                index.append(distance.index(132, index[i-1] + 1))
        # find center of largest patch of 132 cm values (i.e., out of range)
        if len(index) > 0:
            for i in range(len(index)-1):
                if index[i] + 1 == index[i+1]:
                    count += 1
                    if count > max_count:
                        max_count = count
                        center = index[(i+1)-int(count/2)]
                else:
                    count = 0
            if max_count == 0:
                center = index[0]
            return angle[center], distance[center]
        # if there are no values out of range, return the maximum of the list
        else:
            max_index = distance.index(max(distance))
            return angle[max_index], distance[max_index]

        the list will be returned.

        Examples
        --------

        Create an instance of PiBOT.

        >>> from pibot import PiBOT
        >>> robot = PiBOT()

        Command a 360 degree clockwise scan and use the returned values
        to find the minimum.

        >>> angle, distance = robot.scan(-360)
        >>> robot.min_distance(angle, distance)
        (-272.974, 18.2)

        The arguments can also be the lidar data taken directly from the
        PiBOT attributes instead of local copies used in the previous
        example.

        >>> robot.scan(-360)
        ([0.0, -2.713043, -5.634782, -7.721738, -10.22609, -12.83478,...

        min_index = distance.index(min(distance))

[docs]    def detect_objects(self, angle, distance, threshold=10):
        """Attempts to detect one or more objects in the foreground.

        Parameters
        ----------
        angle : list
            The lidar angle data
        distance : list
            The lidar distance data
        threshold : int, float, default=10
            Minimum step size in cm between adjacent lidar distance
            values to indicate an edge is present. 

        Returns
        -------
        3-tuple of lists
            A triplet of lists comprising the approximate heading angles
            in degrees to the center of the objects, the average
            distances in cm to the objects, and their approximate widths
            in cm.

        Notes
        -----
        Attempts to find objects in the foreground by calculating the
        difference between adjacent lidar distance values to detemine
        if there is a step larger than a threshold value. The step
        represents an edge between a near and distant object. If both a
        step from far to near and then a corresponding step from near to
        far are detected, an object is recorded. The center angle,
        average distance, and approximate width of the object are stored
        in the corresponsing lists. After finding the first object,
        detection continues until reaching the end of the data list. If
        the object is too far away or too narrow, it may not be
        detected. It's also possible that this method will return data
        for an object that doesn't really exist. Adjusting the threshold
        value could help fine tune the performance.

        Example
        -------

        Create an instance of PiBOT.

        >>> from pibot import PiBOT
        >>> robot = PiBOT()

        Command a 360 degree clockwise scan and use the returned values
        to detect objects. In this case the unpacking operator is used
        to pass the angle and distance lists returned from the scan
        directly into the .detect_objects() method. The returned object
        lists are stored as local variables as shown. In this example
        three objects are detected.

        >>> obj_angle, obj_dist, obj_width = robot.detect_objects(*robot.scan(-360))
        >>> obj_angle
        [-119.113, -192.1043, -253.1478]
        >>> obj_dist
        [35.8, 64.3, 23.9]
        >>> obj_width
        [37.8, 10.3, 36.0]

        """

        BEAM_ANGLE = 10 # approximate width of lidar beam in degrees
        first_edge = False
        second_edge = False
        object_angle = []
        object_distance = []
        object_width = []
        # compare all adjacent distances for step changes indicating edges
        for i in range(len(distance) - 1):
            delta_distance = distance[i+1] - distance[i]
            # look for first edge where distance steps down from far to near
            if delta_distance < -threshold:
                first_edge = True
                index_first_edge = i+1
            # after finding first edge, look for second where distance steps up
            elif delta_distance > threshold and first_edge and not second_edge:
                second_edge = True
                index_second_edge = i
            # if both edges found, record the center angle, distance, and width
            if first_edge and second_edge:
                center_angle = (angle[index_first_edge]
                                + angle[index_second_edge])/2
                included_angle = abs(angle[index_second_edge]
                                     - angle[index_first_edge])
                # adjust for beam width
                if included_angle > BEAM_ANGLE:
                    included_angle = included_angle - BEAM_ANGLE
                else:
                    included_angle = 5
                # find the average distance of the object
                average_distance = 0
                for i in range(index_first_edge, index_second_edge):
                    average_distance += distance[i]
                if index_first_edge != index_second_edge:
                    average_distance = average_distance / ((index_second_edge)
                                                           - index_first_edge)
                # calculate the approximate object width using law of cosines
                width = sqrt(distance[index_first_edge]**2
                             + distance[index_second_edge]**2
                             - (2*distance[index_first_edge]
                                *distance[index_second_edge]
                                *cos(radians(included_angle))))
                # add object data to the corresponding lists
                object_angle.append(center_angle)
                object_distance.append(round(average_distance, 1))
                object_width.append(round(width, 1))
                # reset to detect more objects
                first_edge = False
                second_edge = False
        return object_angle, object_distance, object_width
